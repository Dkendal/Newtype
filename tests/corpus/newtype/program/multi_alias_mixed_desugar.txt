Several aliases mixing let, cond, and not each desugar independently

=======

type A as let x = 1, y = 2 in [x, y]

type B as cond do
    a <: b -> c,
    d <: e -> f,
    else -> g
end

type C as if not (a <: b) then c else d end

=======

type A as [1, 2]

type B as
    if a <: b then c
    else
        if d <: e then f
        else g
        end
    end

type C as if a <: b then d else c end
