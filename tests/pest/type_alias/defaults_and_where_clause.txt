With defaults and where clause

=======

type A(x)
  defaults x = number
  where x <: number
  as x

=======

(program
  (statement
    (type_alias
      (ident: "A")
      (type_parameters
        (ident: "x")
      )
      (defaults_caluse
        (type_parameter_default
          (ident: "x")
          (expr
            (primitive
              (primitive_number: "number")
            )
          )
        )
      )
      (where_clause
        (type_constraint
          (ident: "x")
          (extends: "<:")
          (expr
            (primitive
              (primitive_number: "number")
            )
          )
        )
      )
      (expr
        (ident: "x")
      )
    )
  )
)
