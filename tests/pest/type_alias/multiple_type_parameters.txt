With multiple type parameters

=======

type Pair(x, y) as [x, y]

=======

(program
  (statement
    (type_alias
      (ident: "Pair")
      (type_parameters
        (ident: "x")
        (ident: "y")
      )
      (expr
        (tuple
          (expr
            (ident: "x")
          )
          (expr
            (ident: "y")
          )
        )
      )
    )
  )
)
